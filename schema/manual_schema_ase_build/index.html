    Schema for building atomic structures¤

    The schema to configure parameters for building atomic structures using ASE.

    Schema:¤

    ### Schema for building atomic structures using ASE
### To use in when build structures from scratch.

### REF:
# See https://wiki.fysik.dtu.dk/ase/ase/build/build.html# for details of ase.build functions.


structure_type:                 # bulk, molecule, surface,
  type: string
chem_formula:                   # chemical formula/element. e.g., "H2O", "Mg2O2", "Mg",
  type: string
ase_build_arg:                  # [ASE kwargs](https://wiki.fysik.dtu.dk/ase/ase/build/build.html#) for building the structure. The parameters depend on the `structure_type`.
  type: dict
  required: True
    ### Example:
    # ase_build_arg:
    #   crystalstructure: "bcc" # choices: sc,fcc,bcc,tetragonal,bct,hcp,rhombohedral,orthorhombic,mcl,diamond,zincblende,rocksalt,cesiumchloride,fluorite,wurtzite.
    #   a: 3.98                 # lattice constant
    #   cubic: True

supercell:                      # make supercell
  type: list
  default: [1, 1, 1]
pbc:                            # periodic boundary conditions in each direction x, y, z
  type: list
  default: [True, True, True]
set_vacuum:                     # replaces existing vacuum with new vacuum thickness along cell vectors a, b, c. Notes: It is a total vacuum on both sides.
  type: list

    Example config 1:¤

    ##### Build Si bulk structure
from_scratch_Si:
  structure_type: "bulk" # bulk, molecule, surface,
  chem_formula: "Si"     # chemical formula/element. e.g., "H2O", "Mg2O2", "Mg",
  pbc: [1, 1, 1]
  ase_build_arg:
    crystalstructure: "diamond"
    a: 5.43


##### Build MoS2 bulk
from_scratch_mx2_1:
  structure_type: "mx2"
  chem_formula: "MoS2"
  supercell: [ 2, 2, 1 ]
  pbc: [1, 1, 1]
  ase_build_arg:
    kind: '1T'           # '2H' or '1T' for MX2 structure
    a: 3.18
    thickness: 3.19
    vacuum: 1.6          # This is 1/2 distance between `two nearest chalcogen layers` in bulk MX2


##### Build MoS2 monolayer
from_scratch_mx2_2:
  structure_type: "mx2"
  chem_formula: "MoS2"
  supercell: [ 2, 2, 1 ]
  pbc: [1, 1, 1]
  set_vacuum: [0, 0, 25] # total vacuum on both sides
  ase_build_arg:
    kind: '1T'           # '2H' or '1T' for MX2 structure
    a: 3.18
    thickness: 3.19
    vacuum: 1.6


##### Build MoS2 double-layer
from_scratch_33:
  structure_type: "mx2"
  chem_formula: "MoS2"
  supercell: [ 2, 2, 2 ]
  pbc: [1, 1, 1]
  set_vacuum: [0, 0, 25]
  ase_build_arg:
    kind: '1T'           # '2H' or '1T' for MX2 structure
    a: 3.18
    thickness: 3.19
    vacuum: 1.6          # This is 1/2 distance between `two nearest chalcogen layers` in bulk MX2


##### Build Graphene monolayer
from_scratch_4:
  structure_type: "graphene"
  chem_formula: "C2"
  supercell: [ 3, 3, 1 ]
  pbc: [1, 1, 1]
  set_vacuum: [0, 0, 20]
  ase_build_arg:
    a: 2.46
    thickness: 3.35


##### Build Graphite bulk
from_scratch_5:
  structure_type: "graphene"
  chem_formula: "C2"
  supercell: [ 3, 3, 2 ]
  pbc: [1, 1, 1]
  ase_build_arg:
    a: 2.46
    thickness: 3.35
